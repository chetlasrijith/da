a) Sequence, Mean and Sum

This program demonstrates creation of sequences using range(), calculation of mean using NumPy, and sum using built-in functions. 

import numpy as np

# Sequence from 20 to 50
sequence = list(range(20, 51))
print("Sequence of numbers from 20 to 50:", sequence)

# Mean from 20 to 60
mean_20_to_60 = np.mean(range(20, 61))
print("Mean of numbers from 20 to 60:", mean_20_to_60)

# Sum from 51 to 91
sum_51_to_91 = sum(range(51, 92))
print("Sum of numbers from 51 to 91:", sum_51_to_91)

Output:
Mean of numbers from 20 to 60: 40.0
Sum of numbers from 51 to 91: 2911

Odd, Even and Multiples


# Odd numbers from 10 to 36
odd_seq = [num for num in range(10, 37) if num % 2 != 0]
print("Odd numbers:", odd_seq)

# Even numbers from 10 to 36
even_seq = [num for num in range(10, 37) if num % 2 == 0]
print("Even numbers:", even_seq)

# Multiples of 5 from 5 to 50
multiples_5 = [num for num in range(5, 51) if num % 5 == 0]
print("Multiples of 5:", multiples_5)

Mean of Odd and Even Numbers


# Mean of odd numbers (1 to 30)
odd_numbers = [num for num in range(1, 31) if num % 2 != 0]
mean_odd = sum(odd_numbers) / len(odd_numbers)
print("Mean of odd numbers:", mean_odd)

# Mean of even numbers (1 to 30)
even_numbers = [num for num in range(1, 31) if num % 2 == 0]
mean_even = sum(even_numbers) / len(even_numbers)
print("Mean of even numbers:", mean_even)

b) Fibonacci Series

Fibonacci sequence: Each number is the sum of the previous two numbers. 

# First 10 Fibonacci numbers
n = 10
n1, n2 = 0, 1
print(n1)
print(n2)

for i in range(3, n+1):
    new = n1 + n2
    print(new)
    n1, n2 = n2, new

c) Prime Numbers

A prime number has exactly two factors: 1 and itself. 

# Check if a number is prime
n = int(input("Enter n: "))
count = 0

for i in range(1, n+1):
    if n % i == 0:
        count += 1

if count == 2:
    print(n, "is prime")
else:
    print(n, "is not prime")
